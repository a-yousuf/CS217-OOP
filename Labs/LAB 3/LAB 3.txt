        double salary;
        public:
                void setfname(string fname){
                        this -> fname = fname;
                }
                void setlname(string lname){
                        this -> lname = lname;
                }
                void setsalary(double salary){
                        this -> salary = salary;
                }
                string getf(){
                        return fname;
                }
                string getl(){
                        return lname;
                }
                double getsal(){
                        return salary;
                }        
                int raise(){
                        int raise= salary+(salary*0.1);
                        return raise;
                }
                emp(){
                        salary=0.0;        
                }
};
int main(){
        emp ob1;
        emp ob2;
        ob1.setfname("A");
        ob1.setlname("B");
