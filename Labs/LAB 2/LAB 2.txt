float dist(float x1,float x2,float y1,float y2){
        float dist;
        dist = sqrt((pow(x2 - x1, 2)) + (pow(y2 - y1, 2)));
        return dist;
}
float rad(float cx,float cy,float x,float y){
        float rad;
        rad= (dist(x,cx,y,cy));
        return rad;
}
float circumference(float rad){
        float c;
        c=(rad*pi*2);
        return c;
}
float area(float rad){
